        let updateInterval;

        async function loadData() {
            if (isUpdating) return;
            isUpdating = true;
            showRefreshIndicator();

            try {
                console.log('üîÑ Loading data...');
                
                // Lade Status
                console.log('üìä Fetching /api/status...');
                const statusResponse = await fetch('/api/status');
                console.log('üìä Status response:', statusResponse.status, statusResponse.statusText);
                const status = await statusResponse.json();
                console.log('üìä Status data:', status);
                updateStats(status);

                // Lade Teams
                console.log('üêï Fetching /api/teams...');
                const teamsResponse = await fetch('/api/teams');
                console.log('üêï Teams response:', teamsResponse.status, teamsResponse.statusText);
                const teams = await teamsResponse.json();
                console.log('üêï Teams data:', teams);
                displayTeams(teams);

                // Lade Notizen
                console.log('üìã Fetching /api/notes...');
                const notesResponse = await fetch('/api/notes');
                console.log('üìã Notes response:', notesResponse.status, notesResponse.statusText);
                
                if (!notesResponse.ok) {
                    console.error('‚ùå Notes fetch failed:', notesResponse.status, notesResponse.statusText);
                    throw new Error(`Notes fetch failed: ${notesResponse.status}`);
                }
                
                const notesText = await notesResponse.text();
                console.log('üìã Notes raw text:', notesText.substring(0, 200));
                
                let notes;
                try {
                    notes = JSON.parse(notesText);
                    console.log('üìã Notes parsed successfully:', notes);
                } catch (parseError) {
                    console.error('‚ùå JSON parse error:', parseError);
                    console.error('‚ùå Failed text:', notesText);
                    throw parseError;
                }
                
                console.log('üìã Calling displayNotes with:', notes);
                displayNotes(notes);

                hideRefreshIndicator();
                console.log('‚úÖ All data loaded successfully');
            } catch (error) {
                console.error('‚ùå Error loading data:', error);
                console.error('‚ùå Error stack:', error.stack);
                showError('Verbindungsfehler. √úberpr√ºfe die Netzwerkverbindung.');
                hideRefreshIndicator();
            }

            isUpdating = false;
        }

        }

        function displayNotes(notes) {

            console.log('üìã Notes is array:', Array.isArray(notes));
            console.log('üìã Notes length:', notes?.length);

                console.error('‚ùå notesContainer not found in DOM!');
                return;
            }
            
            if (!notes || !Array.isArray(notes) || notes.length === 0) {

                        <div class='empty-state-text'>

        // Reply dialog functionality for mobile
        function showReplyDialog(noteId, originalContent, teamName) {
            const replyText = prompt(`Antwort auf: "${originalContent.substring(0, 50)}..."\n\nDeine Antwort:`);
            
            if (replyText && replyText.trim()) {
                createReply(noteId, replyText.trim(), teamName);
            }
        }

        // Create reply via API
        function createReply(noteId, content, teamName) {
            const replyData = {
                content: content,
                teamName: teamName || null,
                entryType: 'Manual'
            };

            fetch(`/api/notes/${noteId}/reply`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(replyData)
            })
            .then(response => response.json())
            .then(data => {
                if (data.error) {
                    alert('Fehler beim Erstellen der Antwort: ' + data.error);
                } else {
                    console.log('‚úÖ Reply created successfully:', data);
                    // Refresh data to show new reply
                    loadData();
                }
            })
            .catch(error => {
                console.error('‚ùå Error creating reply:', error);
                alert('Fehler beim Erstellen der Antwort');
            });
        }
        
        function showError(message) {
            const container = document.getElementById('teamsContainer');
            container.innerHTML = `<div class='error'>${escapeHtml(message)}</div>`;
        }

        function showRefreshIndicator() {
            document.getElementById('refreshIndicator').style.display = 'flex';
        }

        function hideRefreshIndicator() {
            setTimeout(() => {
                document.getElementById('refreshIndicator').style.display = 'none';
            }, 500);
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }

            if (!isPulling) return;
            const currentY = e.touches[0].pageY;
            if (currentY - startY > 100) {
                loadData();
                isPulling = false;
            }
        });

        document.addEventListener('touchend', () => {
            isPulling = false;
        });

        // Reload when coming back to the page
        document.addEventListener('visibilitychange', () => {
            if (!document.hidden) {
                loadData();
            }
        });

        // Log for debugging
        console.log('Einsatz√ºberwachung Mobile v1.7 loaded');
        console.log('Auto-refresh: every 5 seconds');
    </script>
